\section*{Exercice 1 — Contrôle qualité en menuiserie \pts{5 points}}
% ================================================================

\begin{contextebox}
\textbf{Données :} $P(D) = 0{,}08$ \quad $P(\overline{D}) = 0{,}92$ \quad $P_D(S) = 0{,}3$ \quad $P_{\overline{D}}(S) = 0{,}05$
\end{contextebox}

\vspace{6pt}

% --- Q1 ---
\textbf{Q1} \comp{APP} — Traduire les données en probabilités. \pts{1 pt}

\begin{corrbox}
D'après l'énoncé :
\begin{itemize}[nosep]
  \item $P(D) = 0{,}08$ \quad (8\% des panneaux ont un défaut de dimensions)
  \item $P(\overline{D}) = 1 - P(D) = 1 - 0{,}08 = \boxed{0{,}92}$
  \item $P_D(S) = 0{,}3$ \quad (probabilité de défaut de surface sachant défaut de dimensions)
  \item $P_{\overline{D}}(S) = 0{,}05$ \quad (probabilité de défaut de surface sachant pas de défaut de dimensions)
\end{itemize}

\textit{Barème : 0,25 pt par valeur correcte.}
\end{corrbox}

% --- Q2 ---
\textbf{Q2} \comp{REA} — Compléter l'arbre de probabilités. \pts{1 pt}

\begin{corrbox}
\begin{center}
\begin{tikzpicture}[scale=1, every node/.style={font=\small}]
  \node[circle, fill=black, inner sep=1.5pt] (R) at (0,0) {};
  % Niveau 1
  \node (D) at (3,1.5) {$D$};
  \node (Db) at (3,-1.5) {$\overline{D}$};
  \draw[-] (R) -- (D) node[midway, above, sloped] {$0{,}08$};
  \draw[-] (R) -- (Db) node[midway, below, sloped] {$0{,}92$};
  % Niveau 2 depuis D
  \node (DS) at (6.5,2.3) {$S$};
  \node (DSb) at (6.5,0.7) {$\overline{S}$};
  \draw[-] (D) -- (DS) node[midway, above, sloped] {$0{,}3$};
  \draw[-] (D) -- (DSb) node[midway, below, sloped] {$0{,}7$};
  % Niveau 2 depuis Db
  \node (DbS) at (6.5,-0.7) {$S$};
  \node (DbSb) at (6.5,-2.3) {$\overline{S}$};
  \draw[-] (Db) -- (DbS) node[midway, above, sloped] {$0{,}05$};
  \draw[-] (Db) -- (DbSb) node[midway, below, sloped] {$0{,}95$};
\end{tikzpicture}
\end{center}

\textbf{Vérifications :}
\begin{itemize}[nosep]
  \item Nœud racine : $0{,}08 + 0{,}92 = 1$ \checkmark
  \item Nœud $D$ : $0{,}3 + 0{,}7 = 1$ \checkmark \quad ($P_D(\overline{S}) = 1 - 0{,}3 = 0{,}7$)
  \item Nœud $\overline{D}$ : $0{,}05 + 0{,}95 = 1$ \checkmark \quad ($P_{\overline{D}}(\overline{S}) = 1 - 0{,}05 = 0{,}95$)
\end{itemize}

\textit{Barème : 0,5 pt pour les branches niveau 1 + 0,5 pt pour les branches niveau 2.}
\end{corrbox}

% --- Q3 ---
\textbf{Q3} \comp{REA} — Calculer $P(D \cap S)$ et $P(\overline{D} \cap S)$. \pts{1 pt}

\begin{corrbox}
On multiplie les probabilités le long de chaque chemin :

\textbf{Chemin $D$ puis $S$ :}
\[
P(D \cap S) = P(D) \times P_D(S) = 0{,}08 \times 0{,}3 = \boxed{0{,}024}
\]

\textbf{Chemin $\overline{D}$ puis $S$ :}
\[
P(\overline{D} \cap S) = P(\overline{D}) \times P_{\overline{D}}(S) = 0{,}92 \times 0{,}05 = \boxed{0{,}046}
\]

\textit{Barème : 0,5 pt par calcul correct.}
\end{corrbox}

% --- Q4 ---
\textbf{Q4} \comp{ANA} — Calculer $P(S)$ par la formule des probabilités totales. \pts{1 pt}

\begin{corrbox}
La formule des probabilités totales donne :
\[
P(S) = P(D \cap S) + P(\overline{D} \cap S) = 0{,}024 + 0{,}046 = \boxed{0{,}070}
\]

Soit \textbf{7\%} des panneaux ont un défaut de surface.

\smallskip
\textit{Explication :} l'événement $S$ peut se produire par deux chemins dans l'arbre (via $D$ ou via $\overline{D}$). La probabilité totale est la somme des probabilités de ces deux chemins.

\smallskip
\textit{Barème : 0,5 pt pour la formule posée + 0,5 pt pour le résultat.}
\end{corrbox}

% --- Q5 ---
\textbf{Q5} \comp{ANA} — Application et conclusion. \pts{1 pt}

\begin{corrbox}
Sur 500 panneaux :
\[
\text{Nombre attendu de panneaux avec défaut de surface} = 500 \times 0{,}070 = \boxed{35 \text{ panneaux}}
\]

Taux de défaut de surface : $\frac{35}{500} = 7\%$.

Le responsable qualité considère le taux acceptable si $< 5\%$. Or $7\% > 5\%$.

$\Rightarrow$ \textbf{Non}, le taux n'est pas acceptable. Il faudra améliorer le processus de fabrication pour réduire les défauts, en particulier les défauts de surface sur les panneaux ayant déjà un défaut de dimensions ($P_D(S) = 30\%$ est très élevé).

\smallskip
\textit{Barème : 0,5 pt pour le calcul + 0,5 pt pour la conclusion argumentée.}
\end{corrbox}

\section*{Exercice 2 — Optimisation du volume d'un bac de rangement \pts{5 points}}
% ================================================================

\begin{contextebox}
\textbf{Données :} Plaque 40 cm $\times$ 30 cm. Découpe de côté $x$ aux coins.\\
$V(x) = 4x^3 - 140x^2 + 1\,200x$ pour $x \in \;]0\,;\,15[$
\end{contextebox}

\vspace{6pt}

% --- Q1 ---
\textbf{Q1} \comp{REA} — Tracer la courbe à la calculatrice. \pts{1 pt}

\begin{corrbox}
Sur la calculatrice NumWorks (ou GeoGebra), on saisit $f(x) = 4x^3 - 140x^2 + 1200x$ et on trace sur $[0\,;\,15]$.

\begin{center}
\begin{tikzpicture}[scale=0.55]
  % Axes
  \draw[->] (0,0) -- (16,0) node[right] {$x$};
  \draw[->] (0,0) -- (0,12) node[above] {$V$};
  % Graduations x
  \foreach \x in {5,10,15} {
    \draw (\x,0) -- (\x,-0.2) node[below] {\small\x};
  }
  % Graduations y (×100)
  \foreach \y/\label in {2/1000, 4/2000, 6/3000, 8/4000} {
    \draw (0,\y) -- (-0.2,\y) node[left] {\small\label};
  }
  % Courbe V(x) = 4x³ - 140x² + 1200x (échelle y : /500)
  % Points : V(0)=0, V(3.33)≈3252, V(5)=2500, V(10)=200, V(15)=0
  \draw[thick, bleupro, smooth] plot[domain=0:15, samples=50]
    (\x, {(4*\x*\x*\x - 140*\x*\x + 1200*\x)/500});
  % Maximum
  \filldraw[rouge] (3.33,6.5) circle (3pt);
  \node[above right, rouge, font=\small\bfseries] at (3.33,6.5) {max $\approx (3{,}3\,;\,3\,252)$};
  % Zéros
  \filldraw (0,0) circle (2pt);
  \filldraw (15,0) circle (2pt);
\end{tikzpicture}
\end{center}

L'élève doit reproduire cette allure avec les axes gradués et le maximum visible.

\smallskip
\textit{Barème : 0,5 pt pour la forme générale correcte + 0,5 pt pour les axes et le maximum repéré.}
\end{corrbox}

% --- Q2 ---
\textbf{Q2} \comp{APP} — Lecture graphique du maximum. \pts{1 pt}

\begin{corrbox}
Par lecture graphique sur la calculatrice :

Le volume semble maximal pour $x \approx \boxed{3{,}3 \text{ cm}}$ (entre 3 et 4).

La valeur approchée du maximum est $V \approx \boxed{3\,250 \text{ cm}^3}$.

\smallskip
\textit{Barème : 0,5 pt pour la valeur de $x$ (tolérance $\pm 0{,}5$) + 0,5 pt pour la valeur de $V$ (tolérance $\pm 200$).}
\end{corrbox}

% --- Q3 ---
\textbf{Q3} \comp{REA} — Dérivée et résolution. \pts{1 pt}

\begin{corrbox}
\textbf{Dérivée :}
\[
V'(x) = 3 \times 4x^2 - 2 \times 140x + 1\,200 = 12x^2 - 280x + 1\,200
\]

\textbf{Factorisation} (admise) : $V'(x) = 4(3x - 10)(x - 30)$

\textbf{Résolution} $V'(x) = 0$ :
\[
3x - 10 = 0 \quad \Rightarrow \quad x = \frac{10}{3} \approx 3{,}33
\]
\[
x - 30 = 0 \quad \Rightarrow \quad x = 30 \quad \text{(hors de $]0\,;\,15[$)}
\]

Seule solution dans $]0\,;\,15[$ : $\boxed{x = \dfrac{10}{3} \approx 3{,}33 \text{ cm}}$

\smallskip
\textit{Barème : 0,5 pt pour la dérivée + 0,5 pt pour la résolution.}
\end{corrbox}

% --- Q4 ---
\textbf{Q4} \comp{ANA} — Tableau de variations et maximum exact. \pts{1 pt}

\begin{corrbox}
\textbf{Signe de $V'(x) = 4(3x-10)(x-30)$ sur $]0\,;\,15[$ :}

\begin{center}
\begin{tabular}{c|ccccc}
$x$ & $0$ & & $\frac{10}{3}$ & & $15$ \\
\hline
$3x - 10$ & $-$ & $-$ & $0$ & $+$ & $+$ \\
$x - 30$ & $-$ & $-$ & $-$ & $-$ & $-$ \\
\hline
$V'(x) = 4(\ldots)(\ldots)$ & & $+$ & $0$ & $-$ & \\
\hline
$V$ & $0$ & $\nearrow$ & \textbf{max} & $\searrow$ & $0$ \\
\end{tabular}
\end{center}

\textbf{Calcul du maximum :}
\[
V\!\left(\frac{10}{3}\right) = 4 \times \left(\frac{10}{3}\right)^3 - 140 \times \left(\frac{10}{3}\right)^2 + 1\,200 \times \frac{10}{3}
\]
\[
= 4 \times \frac{1\,000}{27} - 140 \times \frac{100}{9} + 4\,000 = \frac{4\,000}{27} - \frac{14\,000}{9} + 4\,000
\]
\[
= \frac{4\,000 - 42\,000 + 108\,000}{27} = \frac{70\,000}{27} \approx \boxed{2\,593 \text{ cm}^3}
\]

\smallskip
\textit{Note : la valeur exacte est $\frac{70\,000}{27} \approx 2\,593$ cm$^3$. La lecture graphique donnait environ 3\,250 — l'écart vient de la difficulté de lecture précise. Les deux ordres de grandeur sont cohérents.}

\textit{Remarque : si l'élève utilise la calculatrice pour évaluer $V(10/3)$, il obtient directement $\approx 2\,593$.}

\smallskip
\textit{Barème : 0,5 pt pour le tableau de variations + 0,5 pt pour le calcul du maximum.}
\end{corrbox}

% --- Q5 ---
\textbf{Q5} \comp{COM} — Conclusion et dimensions. \pts{1 pt}

\begin{corrbox}
Le volume est maximal pour $x = \dfrac{10}{3} \approx 3{,}33$ cm.

\textbf{Dimensions du bac :}
\begin{itemize}[nosep]
  \item Hauteur : $x = \dfrac{10}{3} \approx 3{,}3$ cm
  \item Longueur : $40 - 2x = 40 - \dfrac{20}{3} = \dfrac{100}{3} \approx 33{,}3$ cm
  \item Largeur : $30 - 2x = 30 - \dfrac{20}{3} = \dfrac{70}{3} \approx 23{,}3$ cm
\end{itemize}

\textbf{Volume maximal :} $V_{\max} = \dfrac{70\,000}{27} \approx 2\,593$ cm$^3$

\textbf{Contrainte :} Le menuisier souhaite un bac de plus de 3\,000 cm$^3$. Or $2\,593 < 3\,000$.

$\Rightarrow$ \textbf{La contrainte n'est pas respectée.} Avec une plaque de 40 $\times$ 30 cm, le volume maximal possible est inférieur à 3\,000 cm$^3$. Il faudra utiliser une plaque plus grande.

\smallskip
\textit{Barème : 0,25 pt dimensions + 0,25 pt volume max + 0,5 pt conclusion sur la contrainte.}
\end{corrbox}
